首先对于八皇后问题就是dfs全排列的特殊版，也就是在dfs的过程中对于此位置的y轴  正对角线  斜对角线进行判断，然后进行下去
那么关键就是如何去判断正对角线与斜对角线
```cpp
vector<vector<int>>l1;
vector<int>l2;
int n;
int y[50], dx[50], dy[50];
void dfs(int k) {
	if (l2.size() == n) {
		l1.push_back(l2);
		return;
	}
	for (int i = 1; i <= n; i++) {
		//对于正对角线以及斜对角线的计算
		//结果表示为对角线的标记
		int a1 = k - i + n;
		int b1 = k + i;
		if (!y[i] && !dx[a1] && !dy[b1]) {
			y[i] = 1;
			dx[a1] = 1;
			dy[b1] = 1;
			l2.push_back(i);
			dfs(k + 1);
			y[i] = 0;
			dx[a1] = 0;
			dy[b1] = 0;
			l2.pop_back();
		}
	}
	return;
}
```
也就是计算正对角线以及斜对角线的标记,如果从零开始则要a1=k-i+n-1;
